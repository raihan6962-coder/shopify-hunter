function doPost(e) { var data = JSON.parse(e.postData.contents); var action = data.action; if (action === 'save_lead') { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(['Date','Store Name','URL','Email','Phone','Country','Keyword']); } sheet.appendRow([ new Date(), data.store_name || '', data.url || '', data.email || '', data.phone || '', data.country || '', data.keyword || '' ]); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } if (action === 'send_email') { GmailApp.sendEmail( data.to, data.subject, '', {htmlBody: data.body, name: 'Your Name'} ); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput(JSON.stringify({status:'unknown action'})) .setMimeType(ContentService.MimeType.JSON); }
